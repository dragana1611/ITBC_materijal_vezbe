#define _CRT_SECURE_NO_WARNINGS
#include<stdio.h>
#include<math.h>
int Armstrongov(int br)
{
	int i, S=0, a, b, c;

	a = br / 100;
	b = br / 10 % 10;
	c = br % 10;

	S = pow(a, 3) + pow(b, 3) + pow(c, 3);

	if (S == br)
	{
		return 1;
	}
}
