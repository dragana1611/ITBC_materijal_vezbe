/*Napisati funkciju koja nalazi minimalni ƒçlan niza i njegovu poziciju u nizu. U glavnom programu
uneti matricu i za svaku vrstu prikazati minimalni elemenat, kao i kolonu u kojoj se on nalazi (tj.
njegov indeks u toj vrsti).*/

#include<stdio.h>
#include<stdlib.h>

int MIN(int niz[], int n)
{
	int i = 0;
	int min = niz[0];

	for (i = 0; i < n; i++)
	{
		if (min > niz[i])
		{
			min = niz[i];
		}
	}
	return min;
}

main()
{
	int i, j, mat[30][30], n, m;

	printf("Unesite duzine matrice ");
	scanf_s("%d%d", &n, &m);

	printf("Unesite matricu.\n");
	for (i = 0; i < n; i++)
	{
		for (j = 0; j < m; j++)
		{
			scanf_s("%d", &mat[i][j]);
		}
	}
	
	for (i = 0; i < n; i++)
	{
		printf("%d ", MIN(mat[i], n));
	}
	
}
